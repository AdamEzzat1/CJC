NFA · No Backtracking
~= match · !~ not-match operators 
// ~= returns true if pattern matches
let email = "alice@company.com";
print(email ~= /\w+@\w+\.\w+/);  // true

// !~ returns true if pattern does NOT match
let code = "ABC-123";
print(code !~ /^\d+$/);           // true

// works in conditionals
if email ~= /^admin/ {
    print("admin account");
}
regex inside functions 
fn validate_email(s: String) -> Bool {
    return s ~= /\w+@\w+\.\w+/;
}

print(validate_email("bob@example.org"));
// → true
print(validate_email("not-an-email"));
// → false

fn is_numeric(s: String) -> Bool {
    return s ~= /^\d+(\.\d+)?$/;
}

regex inside TidyView filter & mutate 
// filter rows where email is valid
let clean = df
    .filter(col("email") ~= /\w+@\w+\.\w+/)
    .mutate("domain",
        str_extract(col("email"), /@(\w+\.\w+)/))
    .select(["name", "email", "domain"]);

// result: only valid emails, domain extracted
